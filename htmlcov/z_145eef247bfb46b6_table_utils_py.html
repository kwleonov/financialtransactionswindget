22def read_csv(filename: str) -> list[TransactionData]: 

23 """read_csv(filename) - filename is path to csv file, return list of transaction data from csv file. 

24 csv file must have fields: 

25 id: int, state: str, date: str, amount: float, currency_name: str, currency_code: str, from: str, to: str and 

26 description: str.""" 

27 

28 transaction_data: list[TransactionData] = list() 

29 

30 try: 

31 with open(filename, encoding="utf-8") as csv_file: 

32 csv_data = csv.DictReader(csv_file, delimiter=";") 

33 for row in csv_data: 

34 if not row["id"].isdigit(): 

35 logger.warning(f"incorrect csv data in {filename}: 'id' field must be numeric.") 

36 continue 

37 id = int(row["id"]) 

38 state = row["state"] 

39 date = row["date"] 

40 amount = str(row["amount"]) 

41 currency_name = row["currency_name"] 

42 currency_code = row["currency_code"] 

43 from_val = row["from"] 

44 to_val = row["to"] 

45 description = row["description"] 

46 transaction: TransactionData = { 

47 "id": id, 

48 "state": state, 

49 "date": date, 

50 "operationAmount": { 

51 "amount": amount, 

52 "currency": { 

53 "name": currency_name, 

54 "code": currency_code, 

55 } 

56 }, 

57 "description": description, 

58 "from": from_val, 

59 "to": to_val, 

60 } 

61 transaction_data.append(transaction) 

62 logger.debug("the read_csv function was executed without errors.") 

63 except FileNotFoundError as e: 

64 logger.error(f"CSV file {filename} not found: {e}") 

65 except KeyError as e: 

66 logger.error(f"field name in csv data {filename} not exists: {e}") 

67 except csv.Error as e: 

68 logger.error(f"error parse {filename} csv data: {e}") 

69 

70 return transaction_data
